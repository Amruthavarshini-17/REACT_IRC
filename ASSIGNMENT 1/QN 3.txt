export function calculateRectangleArea(length, width) {
    return length * width;}
import { calculateRectangleArea } from './rectangle.js';
const length = 10;
const width = 5;
const area = calculateRectangleArea(length, width);
console.log(`The area of the rectangle is: ${area}`);
